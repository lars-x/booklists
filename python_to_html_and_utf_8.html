import os

from shutil import copyfile

 

source_dir = r"C:\Gais\1 Original\booklists"

target_dir = r"C:\Gais\2 Klart\booklists"

 

def to_html_and_utf_8(source_name, target_name):

    source_encoding = "iso-8859-1"

    target_encoding = "utf-8"

    source = open(source_name, "r", encoding=source_encoding)

    target = open(target_name, "w", encoding=target_encoding)

    contents  = source.read()

    target.write(contents)

 

dir_list = os.listdir(source_dir)

fileCount = len(dir_list)

for k in range(fileCount):

    source_file_name = dir_list[k]

    source_name = os.path.join(source_dir, source_file_name)

 

    root, ext = os.path.splitext(source_file_name)

    if ext.lower() == ".htm":

        target_name = os.path.join(target_dir, root + ".html")

        to_html_and_utf_8(source_name, target_name)

        print(target_name)

    else:

        target_name = os.path.join(target_dir, source_file_name)

        copyfile(source_name, target_name)
